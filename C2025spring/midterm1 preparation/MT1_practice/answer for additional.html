*********** Problem 2 (remove the BLANK SPACES from a String) ***********
.ORIG x3000
LEA R0, STRING ; R0 contains “Current Read” location
ADD R1, R0, #0 ; R1 contains “Current Write” location
LD R4, SPACE ; R4 contains -32 (minus ASCII for space)
NEXT LDR R2, R0, #0 ; R2 contains current character
ADD R3, R2, R4 ; R3 is a temporary value
BRnp NOTSPACE
ADD R0, R0, #1 ; We have a space
BR NEXT
NOTSPACE STR R2, R1, #0 ; Write to “Current Write” location
ADD R0, R0, #1
ADD R1, R1, #1
ADD R2, R2, #0 ; Test for end of string
BRnp NEXT
DONE
LEA R0, STRING
PUTS
HALT
SPACE .FILL xFFE0 ;-32 in decimal
STRING .STRINGZ "ECE 220 !"
.END
